## 2. 开发规范

### 2.1 TypeScript 编码规范

#### 2.1.1 文件命名规范

```
模块文件：
  - 小写字母，多个单词用连字符分隔
  - 示例: token-manager.ts, auth-interceptor.ts

类文件：
  - 大驼峰命名（PascalCase）
  - 示例: TokenManager.ts, AuthInterceptor.ts

接口文件：
  - types/ 目录下的类型定义文件
  - 使用描述性名称，如 auth.ts, config.ts

测试文件：
  - 与源文件同名，后缀 .test.ts
  - 示例: token-manager.test.ts
```

#### 2.1.2 变量命名规范

```typescript
// 常量：全大写，下划线分隔

#### 2.1.4 函数编写规范

```typescript
/**
 * 获取有效的 Token
 *
 * 自动处理 Token 缓存、过期检查、自动刷新
 *
 * @returns {Promise<string>} 返回有效的 access_token
 * @throws {AuthError} 当获取 Token 失败时抛出
 *
 * @example
 * ```typescript
 * const token = await tokenManager.getValidToken();
 * console.log(token);
 * ```
 */
public async getValidToken(): Promise<string> {
  // 1. 检查缓存
  const cachedToken = await this.loadFromCache();
  if (cachedToken && !this.isExpired(cachedToken)) {
    return cachedToken.access_token;
  }

  // 2. 刷新 Token
  try {
    const newToken = await this.refreshToken();
    await this.saveToCache(newToken);
    return newToken.access_token;
  } catch (error) {

      this.logger.debug('错误堆栈:', error.stack);

    process.exitCode = ExitCode.GENERAL_ERROR;

### 3.4 API 错误拦截器

#### `src/api/middleware/error-interceptor.ts`

```typescript
import axios, { AxiosError } from 'axios';
import { AuthError, NetworkError, BusinessError, ExitCode } from '../../services/error';
import { Logger } from '../../services/logger';

/**
 * API 错误拦截器
 */
export class ErrorInterceptor {
  private logger: Logger;

  constructor(logger: Logger) {
    this.logger = logger;
  }

  /**
   * 安装错误拦截器
   */
  public install(axiosInstance: axios.AxiosInstance): void {
    axiosInstance.interceptors.response.use(
      (response) => response,
      (error: AxiosError) => this.handleError(error)
    );
  }

  /**
   * 处理 HTTP 错误
   */
  private async handleError(error: AxiosError): Promise<void> {
    // 401 错误：Token 过期
    if (error.response?.status === 401) {
      throw new AuthError('Token 已过期或无效', {
        status: 401,
        url: error.config?.url,
      });
    }

    // 429 错误：限流
    if (error.response?.status === 429) {
      throw new BusinessError('请求过于频繁，请稍后重试', 429, {
        status: 429,
        retryAfter: error.response.headers['retry-after'],
      });
    }

    // 5xx 错误：服务端错误
    if (error.response?.status >= 500) {
      throw new NetworkError('服务端错误，请稍后重试', {
        status: error.response.status,
        message: error.response.data,
      });
    }

    // 网络错误：连接失败
    if (error.code === 'ECONNREFUSED' || error.code === 'ETIMEDOUT') {
      throw new NetworkError('网络连接失败', {
        code: error.code,
        url: error.config?.url,
      });
    }

    // 其他错误
    throw new NetworkError(error.message, {
      code: error.code,
      url: error.config?.url,
    });
  }
}

import chalk from 'chalk';

/**
 * 日志管理器
 */
export class Logger {
  private verbose: boolean;
  private noColor: boolean;

  constructor(verbose: boolean = false, noColor: boolean = false) {
    this.verbose = verbose;
    this.noColor = noColor;
  }

  /**
   * 调试日志（仅 --verbose 时输出）
   */
  public debug(message: string, ...args: any[]): void {
    if (this.verbose) {
      this.log(LogLevel.DEBUG, message, ...args);
    }
  }

  /**
   * 信息日志
   */
  public info(message: string, ...args: any[]): void {
    this.log(LogLevel.INFO, message, ...args);
  }

  /**
   * 警告日志
   */
  public warn(message: string, ...args: any[]): void {
    this.log(LogLevel.WARN, message, ...args);
  }

  /**
   * 错误日志
   */
  public error(message: string, ...args: any[]): void {
    this.log(LogLevel.ERROR, message, ...args);
  }

  /**
   * 输出日志
   */
  private log(level: LogLevel, message: string, ...args: any[]): void {
    const timestamp = new Date().toISOString();
    const formattedMessage = this.formatMessage(level, message);

    // 非错误日志输出到 stdout，错误日志输出到 stderr
    const stream = level === LogLevel.ERROR ? process.stderr : process.stdout;

    stream.write(`[${timestamp}] ${formattedMessage}\n`);

    if (args.length > 0) {
      stream.write(JSON.stringify(args, null, 2) + '\n');
    }
  }

  /**
   * 格式化日志消息（带颜色）
   */
  private formatMessage(level: LogLevel, message: string): string {
    if (this.noColor) {
      return `[${level.toUpperCase()}] ${message}`;
    }

    const colors = {
      [LogLevel.DEBUG]: chalk.gray,
      [LogLevel.INFO]: chalk.blue,
      [LogLevel.WARN]: chalk.yellow,
      [LogLevel.ERROR]: chalk.red,
    };

    const color = colors[level] || chalk.white;
    return color(`[${level.toUpperCase()}] ${message}`);
  }

  /**
   * 检查是否启用详细日志
   */
  public isVerbose(): boolean {
    return this.verbose;
  }

  /**
   * 设置日志级别
   */
  public setVerbose(verbose: boolean): void {
    this.verbose = verbose;
  }

  /**
   * 设置是否禁用颜色
   */
  public setNoColor(noColor: boolean): void {
    this.noColor = noColor;
  }
}

### 5.1 Token 管理实现

#### `src/services/auth/token-manager.ts`

```typescript
import { TokenInfo, AuthConfig } from '../../types/auth';
import { FileStorage } from '../../infrastructure/storage';
import { Logger } from '../logger';
import { AuthError, ExitCode } from '../error';
import axios from 'axios';

const TOKEN_EXPIRY_MARGIN = 300; // 提前 5 分钟过期

/**
 * Token 管理器
 *
 * 负责 Token 的获取、刷新、缓存管理
 */
export class TokenManager {
  private storage: FileStorage;
  private logger: Logger;
  private authConfig: AuthConfig;

  constructor(storage: FileStorage, logger: Logger, authConfig: AuthConfig) {
    this.storage = storage;
    this.logger = logger;
    this.authConfig = authConfig;
  }

  /**
   * 获取有效的 Token
   *
   * 自动处理缓存、过期、刷新
   */
  public async getValidToken(): Promise<string> {
    // 1. 检查缓存
    const cachedToken = await this.loadFromCache();
    if (cachedToken) {
      this.logger.debug('从缓存加载 Token');

      // 2. 检查是否过期
      if (!this.isExpired(cachedToken)) {
        this.logger.debug('Token 有效，直接使用');
        return cachedToken.access_token;
      }

      this.logger.debug('Token 已过期，需要刷新');
    }

    // 3. 刷新 Token
    try {
      const newToken = await this.refreshToken();
      await this.saveToCache(newToken);
      this.logger.info('Token 已刷新');
      return newToken.access_token;
    } catch (error) {
      this.logger.error('Token 刷新失败:', error);
      throw new AuthError('Token 获取失败，请检查 AK/SK 配置', ExitCode.AUTH_ERROR);
    }
  }

  /**
   * 刷新 Token
   */
  private async refreshToken(): Promise<TokenInfo> {
    this.logger.debug('调用 /token API 刷新 Token');

    const baseURL = this.getBaseUrl();
    const response = await axios.post(`${baseURL}/token`, {
      ak: this.authConfig.ak,
      sk: this.authConfig.sk,
    });

    const { access_token, expires_in } = response.data;

    // 计算过期时间（提前 5 分钟）
    const expires_at = Date.now() + expires_in * 1000 - TOKEN_EXPIRY_MARGIN * 1000;

    return {
      access_token,
      expires_at,
      token_type: 'Bearer',
    };
  }

  /**
   * 从缓存加载 Token
   */
  private async loadFromCache(): Promise<TokenInfo | null> {
    try {
      const tokenData = await this.storage.read('token.json');
      return JSON.parse(tokenData);
    } catch (error) {
      this.logger.debug('缓存不存在或读取失败');
      return null;
    }
  }

  /**
   * 保存 Token 到缓存
   */
  private async saveToCache(token: TokenInfo): Promise<void> {
    const tokenData = JSON.stringify(token, null, 2);
    await this.storage.write('token.json', tokenData, 0o600); // 权限 600
    this.logger.debug('Token 已保存到缓存');
  }

  /**
   * 检查 Token 是否过期
   */
  private isExpired(token: TokenInfo): boolean {
    return Date.now() >= token.expires_at;
  }

  /**
   * 清除 Token 缓存
   */
  public async clearCache(): Promise<void> {
    await this.storage.delete('token.json');
    this.logger.info('Token 缓存已清除');
  }

  /**
   * 获取 API BaseURL
   */
  private getBaseUrl(): string {
    const urls = {
      sandbox: 'https://openapi-sit.yonyoucloud.com',
      production: 'https://openapi.yonyoucloud.com',
    };
    return urls[this.authConfig.env] || urls.sandbox;
  }
}

### 5.3 命令自动生成实现

#### `src/services/command/generator.ts`

```typescript
import { Command } from 'commander';
import { OpenAPISpec } from '../../types/api';
import { Logger } from '../logger';

/**
 * 命令自动生成器
 *
 * 根据 OpenAPI 规范自动生成 commander 子命令
 */
export class CommandGenerator {
  private logger: Logger;

  constructor(logger: Logger) {
    this.logger = logger;
  }

  /**
   * 根据 OpenAPI 规范生成命令
   */
  public generateFromSpec(program: Command, spec: OpenAPISpec): void {
    this.logger.info('开始生成命令...');

    // 遍历所有 paths
    for (const [path, pathItem] of Object.entries(spec.paths)) {
      this.generatePathCommands(program, path, pathItem);
    }

    this.logger.info(`命令生成完成，共生成 ${this.countCommands(program)} 个命令`);
  }

  /**
   * 为单个 API path 生成命令
   */
  private generatePathCommands(
    program: Command,
    path: string,
    pathItem: any
  ): void {
    // 遍历所有操作（get, post, put, delete）
    for (const [method, operation] of Object.entries(pathItem)) {
      if (['get', 'post', 'put', 'delete', 'patch'].includes(method)) {
        this.generateOperationCommand(program, path, method, operation);
      }
    }
  }

  /**
   * 为单个操作生成命令
   */
  private generateOperationCommand(
    program: Command,
    path: string,
    method: string,
    operation: any
  ): void {
    // 解析命令名称
    const { domain, action } = this.parseCommandName(operation);

    // 创建或获取域命令
    let domainCmd = program.commands.find((cmd) => cmd.name() === domain);
    if (!domainCmd) {
      domainCmd = program.command(domain).description(`${domain} 业务域`);
    }

    // 创建操作命令
    const actionCmd = domainCmd.command(action);

    // 设置描述
    actionCmd.description(operation.summary || `${domain} ${action}`);

    // 添加参数选项
    if (operation.parameters) {
      for (const param of operation.parameters) {
        this.addParameterOption(actionCmd, param);
      }
    }

    // 设置 action 处理函数
    actionCmd.action(async (options) => {
      await this.executeOperation(path, method, operation, options);
    });

    this.logger.debug(`生成命令: ${domain} ${action}`);
  }

  /**
   * 解析命令名称
   *
   * 根据 operation.tags 和 operationId 解析
   */
  private parseCommandName(operation: any): { domain: string; action: string } {
    const tag = operation.tags?.[0] || 'default';
    const operationId = operation.operationId || 'unknown';

    // 示例: operationId = "staff_query" -> domain="staff", action="query"
    const parts = operationId.split('_');
    const domain = parts[0] || tag;
    const action = parts[1] || operationId;

    return { domain, action };
  }

  /**
   * 添加参数选项
   */
  private addParameterOption(cmd: Command, param: any): void {
    const optionName = `--${param.name}`;
    const description = param.description || param.name;

    if (param.required) {
      cmd.requiredOption(optionName, description);
    } else {
      cmd.option(optionName, description);
    }
  }

  /**
   * 执行操作
   */
  private async executeOperation(
    path: string,
    method: string,
    operation: any,
    options: any
  ): Promise<void> {
    // TODO: 实际 API 调用逻辑
    this.logger.info(`执行命令: ${method.toUpperCase()} ${path}`);
    this.logger.debug('参数:', options);
  }

  /**
   * 统计命令数量
   */
  private countCommands(program: Command): number {
    let count = program.commands.length;
    for (const cmd of program.commands) {
      count += cmd.commands.length;
    }
    return count;
  }
}
```

import { JsonFormatter } from './json';

/**
 * 统一输出适配器
 */
export class OutputFormatter {
  private logger: Logger;
  private jsonFormatter: JsonFormatter;
  private tableFormatter: TableFormatter;
  private csvFormatter: CsvFormatter;
  private rawFormatter: RawFormatter;

  constructor(logger: Logger) {
    this.logger = logger;
    this.jsonFormatter = new JsonFormatter();
    this.tableFormatter = new TableFormatter();
    this.csvFormatter = new CsvFormatter();
    this.rawFormatter = new RawFormatter();
  }

  /**
   * 格式化输出数据
   */
  public format(data: any, options: OutputOptions): void {
    try {
      // RAW 模式优先级最高
      if (options.raw) {
        this.rawFormatter.format(data);
        return;
      }

      // 根据格式选项选择格式化器
      switch (options.format) {
        case 'json':
          this.jsonFormatter.format(data);
          break;
        case 'table':
          this.tableFormatter.format(data);
          break;
        case 'csv':
          this.csvFormatter.format(data);
          break;
        default:
          this.logger.warn(`未知格式: ${options.format}, 使用默认 table 格式`);
          this.tableFormatter.format(data);
      }
    } catch (error) {
      this.logger.error('输出格式化失败:', error);
      // 降级为 JSON 输出
      this.jsonFormatter.format(data);
    }
  }
}

import Table from 'cli-table3';
import chalk from 'chalk';

/**
 * Table 格式化器
 */
export class TableFormatter {
  /**
   * 格式化表格输出
   */
  public format(data: any): void {
    // 处理数组数据
    if (Array.isArray(data)) {
      this.formatArray(data);
    } else {
      this.formatObject(data);
    }
  }

  /**
   * 格式化数组数据（表格）
   */
  private formatArray(data: any[]): void {
    if (data.length === 0) {
      console.log(chalk.gray('无数据'));
      return;
    }

    // 提取列名（第一条数据的 keys）
    const keys = Object.keys(data[0]);

    // 创建表格
    const table = new Table({
      head: keys.map((key) => chalk.cyan(key)),
      colWidths: this.calculateColWidths(keys, data),
    });

    // 添加行数据
    for (const row of data) {
      const values = keys.map((key) => this.formatValue(row[key]));
      table.push(values);
    }

    console.log(table.toString());
  }

  /**
   * 格式化对象数据（键值对）
   */
  private formatObject(data: any): void {
    const table = new Table({
      style: { compact: true },
    });

    for (const [key, value] of Object.entries(data)) {
      table.push([chalk.cyan(key), this.formatValue(value)]);
    }

    console.log(table.toString());
  }

  /**
   * 格式化单个值
   */
  private formatValue(value: any): string {
    if (value === null || value === undefined) {
      return chalk.gray('N/A');
    }
    if (typeof value === 'object') {
      return JSON.stringify(value);
    }
    return String(value);
  }

  /**
   * 计算列宽度
   */
  private calculateColWidths(keys: string[], data: any[]): number[] {
    return keys.map((key) => {
      const maxLength = Math.max(
        key.length,
        ...data.map((row) => String(row[key] || '').length)
      );
      // 限制最大宽度 50
      return Math.min(maxLength + 2, 50);
    });
  }
}
